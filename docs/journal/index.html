```{=html}
document.addEventListener('DOMContentLoaded', function() {
  var carnet = window.INEG ? window.INEG.loadCarnet() : {};
  var keys = Object.keys(carnet);
  var total = 12;
  var done = keys.filter(function(k){ return carnet[k] && carnet[k].texte && carnet[k].texte.length > 20; }).length;
  var xp = window.INEG ? window.INEG.loadXP() : { total:0 };

  var grid = document.getElementById('carnet-stats-grid');
  var stats = [
    { label:'Entrées rédigées', val: done + ' / ' + total },
    { label:'Mots écrits (estim.)', val: keys.reduce(function(acc,k){ return acc + (carnet[k]&&carnet[k].texte ? carnet[k].texte.split(/\s+/).filter(Boolean).length : 0); }, 0).toLocaleString('fr-FR') },
    { label:'XP du carnet', val: (done * 20) + ' XP' },
    { label:'Niveau Thomas', val: window.INEG ? window.INEG.getNiveau(xp.total).nom.split(' ').slice(0,2).join(' ') : '—' }
  ];

  stats.forEach(function(s) {
    var div = document.createElement('div');
    div.style.cssText = 'background:#F5F0E8;border-radius:8px;padding:.75rem 1rem;';
    div.innerHTML = '<div style="font-size:.62rem;text-transform:uppercase;letter-spacing:.1em;color:#5A6A7A;margin-bottom:.2rem;">' + s.label + '</div>' +
      '<div style="font-size:1.3rem;font-weight:500;color:#2A3240;">' + s.val + '</div>';
    grid.appendChild(div);
  });

  var bar = document.getElementById('carnet-prog-bar');
  if (bar) bar.style.width = Math.round(done/total*100) + '%';
  var txt = document.getElementById('carnet-prog-txt');
  if (txt) txt.textContent = done + ' / ' + total + ' entrées rédigées';
});
</script>
```

```{=html}
function exportCarnet() {
  var carnet = window.INEG ? window.INEG.loadCarnet() : {};
  var labels = {
    'acte_1_q1': 'Acte I — Perception de l\'inégalité',
    'acte_1_q2': 'Acte I — Les données et le Gini',
    'acte_2_q1': 'Acte II — La Peste comme laboratoire',
    'acte_2_q2': 'Acte II — Inégalité sacrée vs naturelle',
    'acte_3_q1': 'Acte III — Commerce et inégalités',
    'acte_3_q2': 'Acte III — Progrès et inégalité',
    'acte_4_q1': 'Acte IV — Travail et capital',
    'acte_4_q2': 'Acte IV — Penseurs du XIXe siècle',
    'acte_5_q1': 'Acte V — Crise et redistribution',
    'acte_5_q2': 'Acte V — Piketty et r > g',
    'acte_6_q1': 'Acte VI — Inégalités contemporaines',
    'acte_6_q2': 'Acte VI — Que faire ?'
  };
  var lines = ['JOURNAL DE BORD — INÉGALITÉS : HISTOIRE ÉCONOMIQUE', '═'.repeat(50), ''];
  Object.keys(labels).forEach(function(k) {
    var entry = carnet[k];
    lines.push('── ' + labels[k]);
    if (entry && entry.texte) {
      lines.push(entry.texte);
      lines.push('');
    } else {
      lines.push('[Non rédigé]');
      lines.push('');
    }
  });
  var text = lines.join('\n');
  var el = document.getElementById('export-result');
  el.style.display = 'block';
  el.textContent = text;
  navigator.clipboard && navigator.clipboard.writeText(text).then(function() {
    el.textContent = '✓ Copié dans le presse-papier !\n\n' + text;
  });
}
</script>
```
