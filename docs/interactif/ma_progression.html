  ];

  var PROFILS = {
    a: { nom:'Conservateur de l\'ordre', auteurs:'Aristote · Burke · Hayek',
         desc:'Tu tends à accepter les hiérarchies existantes comme fonctionnelles. Les inégalités te semblent souvent le prix de la liberté ou de l\'efficacité.', couleur:'#8B2020' },
    b: { nom:'Réformiste institutionnel', auteurs:'Smith · Keynes · Piketty',
         desc:'Tu cherches à corriger les excès sans renverser le système. Les inégalités sont un problème à gérer, pas une fatalité.', couleur:'#C8962A' },

  var carnet = window.INEG.loadCarnet();

  var quizDone = Object.keys(quizzes).length;
  document.getElementById('stat-xp-total').textContent = xpData.total.toLocaleString('fr-FR');
  document.getElementById('stat-quiz-done').textContent = quizDone + ' / 7';
  document.getElementById('stat-carnet-done').textContent = carnetDone + ' / 12';

});

function resetProg() {
  localStorage.removeItem('ineg_xp');
  localStorage.removeItem('ineg_carnet');
  localStorage.removeItem('ineg_decisions');
  localStorage.removeItem('ineg_quiz');
  window.location.reload();
}
