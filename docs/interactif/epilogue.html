  function selectProfil(decisions) {
    const counts = { a: 0, b: 0, c: 0 };
    Object.values(decisions).forEach(d => {
      if (d.choice && counts[d.choice] !== undefined) counts[d.choice]++;
    });

    const total = counts.a + counts.b + counts.c;
    const pa = counts.a / total;
    const pb = counts.b / total;
    const pc = counts.c / total;

    // Seuil de dominance : >= 4/6
    if (counts.a >= 4) return 'a';
    if (counts.b >= 4) return 'b';
    if (counts.c >= 4) return 'c';

    // Mixtes : les deux composantes >= 2, la troisième <= 1
    if (counts.a >= 2 && counts.b >= 2 && counts.c <= 1) return 'ab';
    if (counts.b >= 2 && counts.c >= 2 && counts.a <= 1) return 'bc';
    if (counts.a >= 2 && counts.c >= 2 && counts.b <= 1) return 'ac';

    // Sinon : équilibré
    return 'equilibre';
  }

    }
  }

  function init() {

    const nb = Object.keys(decisions).length;
    document.getElementById('epi-nb-dec').textContent = nb;
