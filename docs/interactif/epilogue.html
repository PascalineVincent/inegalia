<script>
(function() {
  const STORAGE_KEY = 'voyage_inegalites_decisions';

  const ACTE_LABELS = {
    1: 'Acte I — Antiquité',
    2: 'Acte II — Moyen Âge',
    3: 'Acte III — Renaissance',
    4: 'Acte IV — XIXe s.',
    5: 'Acte V — XXe s.',
    6: 'Acte VI — XXIe s.'
  };

  function loadDecisions() {
    try { return JSON.parse(localStorage.getItem(STORAGE_KEY) || '{}'); } catch(e) { return {}; }
  }

  function buildPrompt(decisions) {
    const lines = [];
    for (let i = 1; i <= 6; i++) {
      const d = decisions['acte_' + i];
      if (d) lines.push(`- ${ACTE_LABELS[i]} : posture "${d.label}" — "${d.echo}"`);
    }
    return `Tu es un historien de la pensée économique qui rédige un épilogue personnalisé pour un étudiant ayant traversé 6 actes d'un cours sur les inégalités économiques (de l'Antiquité au XXIe siècle).

Voici les 6 décisions prises par l'étudiant dans les dilemmes historiques :

${lines.join('\n')}

Rédige un épilogue en 4 parties clairement séparées par des titres :

1. **Ta tradition de pensée** — Identifie la famille intellectuelle dominante de l'étudiant (aristotélicienne/conservatrice, réformiste/institutionnaliste, critique/subversive) en t'appuyant sur les décisions concrètes. Sois précis, pas vague.

2. **Tes tensions internes** — Repère les contradictions ou évolutions dans les choix (ex : conservateur au Moyen Âge, subversif au XIXe). Qu'est-ce que ça révèle ?

3. **Tes références à explorer** — Suggère 3 à 4 auteurs ou œuvres (économistes, philosophes, littéraires) cohérents avec ce profil, en expliquant brièvement pourquoi chacun résonne avec les décisions prises.

4. **La question qui reste ouverte** — Pose une question finale, personnelle, qui découle des tensions de ce parcours et que l'étudiant devrait continuer à explorer.

Ton style : sobre, intellectuel, en français. Évite les formules génériques. Tutoie l'étudiant. Chaque partie fait 4 à 6 phrases. Commence directement par le premier titre sans introduction.`;
  }

  async function generate(decisions, regen) {
    const btn = document.getElementById('epi-btn');
    const loading = document.getElementById('epi-loading');
    const result = document.getElementById('epi-result');
    const body = document.getElementById('epi-result-body');

    btn.disabled = true;
    loading.classList.add('visible');
    result.classList.remove('visible');

    try {
      const resp = await fetch('https://api.anthropic.com/v1/messages', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          model: 'claude-sonnet-4-20250514',
          max_tokens: 1000,
          messages: [{ role: 'user', content: buildPrompt(decisions) }]
        })
      });
      const data = await resp.json();
      const text = data.content?.find(c => c.type === 'text')?.text || '';

      // Convertir le markdown simple en HTML
      const html = text
        .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
        .replace(/^#{1,3} (.+)$/gm, '<h3>$1</h3>')
        .replace(/\n\n/g, '</p><p>')
        .replace(/^/, '<p>').replace(/$/, '</p>');

      body.innerHTML = html;

      // Label profil
      const counts = { a: 0, b: 0, c: 0 };
      Object.values(decisions).forEach(d => { if (counts[d.choice] !== undefined) counts[d.choice]++; });
      const dom = Object.entries(counts).sort((x,y) => y[1]-x[1])[0][0];
      const labels = { a: 'Tradition conservatrice de l\'ordre', b: 'Tradition réformiste institutionnelle', c: 'Tradition critique et subversive' };
      document.getElementById('epi-profil-label').textContent = labels[dom] || '';

      result.classList.add('visible');
    } catch(err) {
      body.innerHTML = '<p style="color:var(--rouge)">Une erreur est survenue. Vérifie ta connexion et réessaie.</p>';
      result.classList.add('visible');
    } finally {
      loading.classList.remove('visible');
      btn.disabled = false;
    }
  }

  function renderRecap(decisions) {
    const recap = document.getElementById('epi-recap');
    recap.innerHTML = '';
    for (let i = 1; i <= 6; i++) {
      const d = decisions['acte_' + i];
      if (!d) continue;
      const el = document.createElement('div');
      el.className = 'epi-recap-item';
      el.innerHTML = `<div class="epi-recap-acte">${ACTE_LABELS[i]}</div>
        <div class="epi-recap-label">${d.label}</div>
        <div class="epi-recap-echo">${d.echo.slice(0, 90)}…</div>`;
      recap.appendChild(el);
    }
  }

  function init() {
    const decisions = loadDecisions();
    const nb = Object.keys(decisions).length;
    document.getElementById('epi-nb-dec').textContent = nb;

    if (nb < 6) {
      document.getElementById('epi-locked').style.display = 'block';
      document.getElementById('epi-ready').style.display = 'none';
    } else {
      document.getElementById('epi-locked').style.display = 'none';
      document.getElementById('epi-ready').style.display = 'block';
      renderRecap(decisions);

      document.getElementById('epi-btn').addEventListener('click', () => generate(decisions, false));
      document.getElementById('epi-regen').addEventListener('click', () => generate(decisions, true));
    }
  }

  init();
  window.addEventListener('storage', init);
})();
</script>
