<script>
var diagAnswers = {};

function showStep(id) {
  document.querySelectorAll('.diag-step').forEach(s => s.classList.remove('active'));
  var el = document.getElementById(id);
  if (el) { el.classList.add('active'); el.scrollIntoView({behavior:'smooth',block:'start'}); }
  if (id === 'step-result') buildResult();
}

function selectOpt(btn, optsId, nextStep) {
  var opts = document.getElementById(optsId);
  if (opts) opts.querySelectorAll('.diag-opt').forEach(b => b.classList.remove('selected'));
  btn.classList.add('selected');
  var q = optsId.replace('opts-','');
  diagAnswers['q' + q] = btn.dataset.val;
  setTimeout(() => showStep(nextStep), 350);
}

function buildResult() {
  var niveau = diagAnswers.q1 || 'intermediaire';
  var piketty = diagAnswers.q2 || 'c';
  var interet = diagAnswers.q3 || 'histoire';
  var attente = diagAnswers.q4 || 'culture';
  var temps   = diagAnswers.q5 || 'moyen';

  var profiles = {
    debutant:      { emoji:'🌱', level:'Niveau découverte', name:'L\'explorateur', desc:'Tu arrives sans a priori. C\'est souvent la meilleure position pour apprendre — tu vas être surpris par des connexions inattendues entre l\'Antiquité et aujourd\'hui.' },
    intermediaire: { emoji:'📚', level:'Niveau consolidation', name:'Le consolidateur', desc:'Tu as des bases. Ce cours va les structurer, les connecter et les challenger avec des données et des débats que tu n\'as peut-être pas encore rencontrés.' },
    avance:        { emoji:'🎓', level:'Niveau critique', name:'L\'analyste', desc:'Tu as des fondations solides. Ce cours va t\'inviter à les remettre en question — surtout sur les angles morts (genre, colonialisme, données) que les manuels standard ignorent.' }
  };

  var p = profiles[niveau] || profiles.intermediaire;
  document.getElementById('result-emoji').textContent = p.emoji;
  document.getElementById('result-level').textContent = p.level;
  document.getElementById('result-name').textContent = p.name;
  document.getElementById('result-desc').textContent = p.desc;

  // Tags
  var tagMap = {histoire:'Histoire longue',donnees:'Données & mesure',philo:'Philosophie éco.',politique:'Enjeux contemporains',culture:'Culture générale',methode:'Méthode critique',argument:'Argumentation',diplome:'Préparation exam'};
  var tags = [tagMap[interet], tagMap[attente]].filter(Boolean);
  document.getElementById('result-tags').innerHTML = tags.map(t => `<span class="diag-tag">${t}</span>`).join('');

  // Recommandations
  var recos = [];
  if (temps === 'peu') {
    recos = [
      { label:'🎯 Priorité', color:'#2563EB', href:'../actes/01_antiquite.html', title:'Les 6 actes (lecture + quiz)', desc:'L\'essentiel en 30 min par acte.' },
      { label:'⚡ Rapide', color:'#059669', href:'../interactif/quiz.html', title:'Quiz global 15 questions', desc:'Évaluer ses connaissances en 20 min.' }
    ];
  } else if (interet === 'donnees' || attente === 'methode') {
    recos = [
      { label:'📊 Recommandé', color:'#059669', href:'../interactif/atelier_donnees.html', title:'Atelier données', desc:'Gini mondial, courbe de Kuznets, r > g.' },
      { label:'🔍 Activité', color:'#7C3AED', href:'../interactif/detective.html', title:'Détective des inégalités', desc:'Identifier une époque par ses données.' }
    ];
  } else if (interet === 'philo' || attente === 'argument') {
    recos = [
      { label:'⚖️ Recommandé', color:'#DC2626', href:'../interactif/debats.html', title:'Débats structurés', desc:'3 débats avec positions argumentées.' },
      { label:'🗺️ Parcours', color:'#2563EB', href:'../actes/01_antiquite.html', title:'Commencer par les décisions Thomas', desc:'Chaque acte a un dilemme philosophique.' }
    ];
  } else {
    recos = [
      { label:'🚀 Par ici', color:'#2563EB', href:'../actes/01_antiquite.html', title:'Acte I — Antiquité', desc:'Le début logique du voyage.' },
      { label:'🏆 Objectif', color:'#D97706', href:'../interactif/ma_progression.html', title:'Ma progression', desc:'Suivre Thomas et débloquer les niveaux.' }
    ];
  }

  document.getElementById('result-recos').innerHTML = recos.map(r => `
    <a href="${r.href}" class="diag-reco-card">
      <div class="diag-reco-label" style="color:${r.color};">${r.label}</div>
      <div class="diag-reco-title">${r.title}</div>
      <div class="diag-reco-desc">${r.desc}</div>
    </a>`).join('');

  // XP
  if (window.INEG) {
    var qd = window.INEG.loadQuiz();
    if (!qd['diagnostic']) {
      qd['diagnostic'] = { ts: Date.now() };
      window.INEG.saveQuiz(qd);
      window.INEG.gagne('diag_complete');
    }
  }
}
</script>
