let cur = 0, score = 0, answered = false;

function buildProg() {
  document.getElementById('prog').innerHTML =
    qs.map((_,i) =>
      `<div class="qcm-dot ${i < cur ? 'done' : i === cur ? 'active' : ''}"></div>`
    ).join('');
}

function buildQ() {
  answered = false;
  const q = qs[cur];
  document.getElementById('qcm-body').innerHTML = `
    <div class="qcm-epoch">${q.epoch}</div>
    <div class="qcm-q">${q.q}</div>
    ${q.opts.map((o,i) =>
      `<button class="qcm-opt" id="opt${i}" onclick="pick(${i})">${o}</button>`
    ).join('')}
    <div id="fb" style="display:none"></div>
    <div class="qcm-nav">
      <button class="qcm-btn primary" id="nxt" onclick="next()" style="display:none">
        ${cur < qs.length - 1 ? 'Question suivante →' : 'Voir le résultat →'}
      </button>
    </div>`;
  buildProg();
}

function pick(i) {
  if (answered) return;
  answered = true;
  const q = qs[cur];
  const opts = document.querySelectorAll('.qcm-opt');
  opts.forEach(o => o.disabled = true);
  opts[i].classList.add(i === q.ans ? 'correct' : 'wrong');
  if (i === q.ans) { score++; } else { opts[q.ans].classList.add('correct'); }
  const fb = document.getElementById('fb');
  fb.className = 'qcm-feedback ' + (i === q.ans ? 'ok' : 'ko');
  fb.innerHTML = '<strong>' + (i === q.ans ? 'Exact.' : 'Pas tout à fait.') + '</strong> ' + q.expl;
  fb.style.display = 'block';
  document.getElementById('nxt').style.display = 'block';
}

function next() {
  cur++;
  if (cur >= qs.length) { showScore(); return; }
  buildQ();
}

function showScore() {
  const pct = Math.round(score / qs.length * 100);
  const msg = pct >= 80 ? "Excellente maîtrise du fil rouge des inégalités."
            : pct >= 60 ? "Bonne base — relisez les actes sur les points manquants."
            : "Les fondamentaux sont à consolider — le journal de bord peut aider.";
  document.getElementById('qcm-body').innerHTML = `
    <div style="padding:1.5rem 0;text-align:center">
      <div class="qcm-score-n">${score}/${qs.length}</div>
      <div class="qcm-score-d">${pct}% &nbsp;·&nbsp; ${msg}</div>
      <div class="qcm-score-links">
        Retour vers les actes :
        <a href="../actes/01_antiquite.html">Antiquité</a> ·
        <a href="../actes/04_industrie.html">XIXe s.</a> ·
        <a href="../actes/05_regulation.html">XXe s.</a>
      </div>
      <br>
      <button class="qcm-btn primary" onclick="restart()">Recommencer</button>
    </div>`;
  document.getElementById('prog').innerHTML =
    qs.map(() => '<div class="qcm-dot done"></div>').join('');
}

function restart() { cur = 0; score = 0; buildQ(); }
buildQ();
</script>
