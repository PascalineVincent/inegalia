function load(){try{return JSON.parse(localStorage.getItem(KEY)||'{}')}catch(e){return{}}}
function save(data){localStorage.setItem(KEY,JSON.stringify(data))}

let answers=load();
let answered=Object.keys(answers).filter(k=>k.startsWith('q')).length;

function updateProgress(){
  answered=Object.keys(answers).filter(k=>k.startsWith('q')).length;
  document.getElementById('av-prog').style.width=Math.round(answered/8*100)+'%';
  document.getElementById('av-prog-txt').textContent=answered+' / 8 questions';
}

function renderAll(){
  const container=document.getElementById('av-questions');
  container.innerHTML='';
  QS.forEach(function(q,i){renderQ(q,i,container)});
  updateProgress();
  if(answered===8&&!document.getElementById('av-result').classList.contains('visible')){
    showResult();
  }
}

function renderQ(q,i,container){
  const saved=answers[q.id];
  const block=document.createElement('div');
  block.className='av-block';
  block.id='block-'+q.id;

  let optsHTML='';
  if(q.type==='qcm'||q.type==='intuition'){
    optsHTML='<div class="av-opts">'+q.opts.map(function(o,j){
      let cls='av-opt';
      if(saved!==undefined){
        if(j===saved) cls+=(q.ans!==null&&j===q.ans)?' correct':(q.ans!==null)?' wrong':' chosen';
        else if(q.ans!==null&&j===q.ans&&saved!==j) cls+=' correct';
      }
      const letters=['A','B','C','D'];
      return`<button class="${cls}" data-qi="${i}" data-oi="${j}"${saved!==undefined?' disabled':''}><span class="av-opt-letter">${letters[j]}</span><span>${o}</span></button>`;
    }).join('')+'</div>';
  } else {
    optsHTML=`<textarea class="av-textarea" id="ta-${q.id}" placeholder="Écris librement…" rows="3"${saved!==undefined?' disabled':''}>${saved||''}</textarea>
    <button class="av-valider" id="val-${q.id}"${saved!==undefined?' disabled':''}>Valider →</button>`;
  }

  const retourVisible=saved!==undefined;
  block.innerHTML=`
    <div class="av-block-header">
      <span class="av-block-num">Question ${i+1}</span>
      <span class="av-block-tag">${q.tag}</span>
    </div>
    <div class="av-block-body">
      <div class="av-block-mise">${q.mise}</div>
      <div class="av-block-q">${q.q}</div>
      ${optsHTML}
      <div class="av-retour${retourVisible?' visible':''}" id="retour-${q.id}">
        <div class="av-retour-label">Note du voyageur</div>
        ${q.retour}
      </div>
    </div>`;
  container.appendChild(block);

  if(q.type==='qcm'||q.type==='intuition'){
    block.querySelectorAll('.av-opt').forEach(function(btn){
      btn.addEventListener('click',function(){
        if(saved!==undefined)return;
        const oi=parseInt(this.dataset.oi);
        answers[q.id]=oi;
        answers['_ts_'+q.id]=Date.now();
        save(answers);
        updateProgress();
        // Re-render ce bloc
        const newBlock=document.createElement('div');
        newBlock.className='av-block';
        newBlock.id='block-'+q.id;
        renderQ(q,i,{appendChild:function(el){block.parentNode.replaceChild(el,block)}});
        if(answered===8)showResult();
      });
    });
  } else {
    const valBtn=document.getElementById('val-'+q.id);
    if(valBtn){
      valBtn.addEventListener('click',function(){
        const ta=document.getElementById('ta-'+q.id);
        const val=ta.value.trim();
        if(!val)return;
        answers[q.id]=val;
        answers['_ts_'+q.id]=Date.now();
        save(answers);
        ta.disabled=true;
        this.disabled=true;
        document.getElementById('retour-'+q.id).classList.add('visible');
        updateProgress();
        if(answered===8)showResult();
      });
    }
  }
}

function showResult(){
  const r=document.getElementById('av-result');
  r.classList.add('visible');

  // Score connaissances (q1,q2,q3)
  const factual=[{id:'q1',ans:1},{id:'q2',ans:1},{id:'q3',ans:1}];
  const score=factual.filter(function(f){return answers[f.id]===f.ans}).length;

  // Profil intuitions (q4,q5,q6)
  const intLabels=[
    ['Pessimiste mondial','Nuancé','Optimiste mondial','Surpris'],
    ['Libéral','Réformiste','Critique','Culturaliste'],
    ['Méritocrate','Systémique','Institutionnaliste','Historique']
  ];
  const intQs=['q4','q5','q6'];
  const profil=intQs.map(function(id,i){
    const v=answers[id];
    return v!==undefined?intLabels[i][v]:'—';
  });

  answers['_score_factuel']=score;
  answers['_profil_intuitions']=profil;
  answers['_completed']=Date.now();
  save(answers);

  r.innerHTML=`
    <div class="av-result-header">
      <div class="av-result-title">Ton profil de départ</div>
      <div class="av-result-sub">Conservé dans ton journal · Comparaison possible à l'épilogue</div>
